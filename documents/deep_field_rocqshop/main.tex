\begin{frame}
\frametitle{The context and the problem}
\begin{itemize}
\item Given the goal \(\frac \pi {\pi ^ 2 + \pi ^ 2} = \frac 1{2 \pi}\)
\begin{itemize}
  \item {\tt field} solves it and requires the proofs of
   \(\pi ^ 2 + \pi ^ 2 \neq 0\) and \(\pi \neq 0\)
\end{itemize}
\item Given the goal \(e^{\left(\frac \pi {\pi ^ 2 + \pi ^ 2}\right)} = e^{\left(\frac 1{2 \pi}\right)}\)
\begin{itemize}
\item {\tt field} fails
\item \(x \mapsto e ^ x\) is a function outside the field fragment
\end{itemize}
\item {\tt ring} has the same problem (for formulas without divisions)
\begin{itemize}
\item \(\pi ^ 2 + \pi \times \pi = 2 \pi ^ 2\) is solved
\item \(ln {\left(\pi ^ 2 + \pi \times \pi\right)} = 
  ln\left(2 \pi ^ 2\right)\) is not
\end{itemize}
\end{itemize}
\end{frame}
\begin{frame}
\frametitle{A study of {\tt ring}}
\begin{itemize}
\item A formal language with just integer constants, \(\times\), \(+\), \(-\),
and variables is defined
\item This language is the ring of polynoms.
\item Polynoms can also be ``normalized'': ordered in variables and powers
\item Expressions outside the ring fragment considered as ``variables''
\item {\em reification}: map formulas to the formal language
\item Normalize into polynomial expressions
\item When normalized expressions are equal, conclude positively
\end{itemize}
\end{frame}
\begin{frame}[fragile]%tp6
\frametitle{Examples of {\tt ring}}
\begin{itemize}
\item \(\pi \times (\pi \times 1) + \pi \times \pi\) is {\tt X * (X * 1) + X * X}\begin{itemize}
\item  {\tt X} instantiated with \(\pi\)
\end{itemize}
\item \(\pi \times (\pi \times 1) + \pi \times \pi = 2 \times \pi \times \pi\) is\\
{\tt X * ( X * 1) + X * X = 2 * X * X}
\begin{itemize}
\item This equality can be solved without knowledge about \(\pi\)
\end{itemize}
\item By contrast \(ln {\left(\pi \times (\pi \times 1) + \pi * \pi\right)} = ln {\left(2 \times (\pi \times (\pi \times 1))\right)}\) is
{\tt X = Y},
\begin{itemize}
\item {\tt X} instanciated with \(ln{\left(\pi \times (\pi \times 1) + \pi * \pi\right)}\)\\
{\tt Y} instanciated with \(ln {\left(2 \times (\pi \times (\pi \times 1))\right)}\)
\item Two different polynomial expressions: failure
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame} % tp 9
\frametitle{Normalize the expressions}
\begin{itemize}
\item Call {\tt ring\_simplify} on all these expressions
\item Fix an order of unknowns : \(x\), \(\cos (x + x)\),  \(\cos (2 x)\), \dots
\begin{itemize}
\item \(\cos (x + \cos (x + x))  \rightarrow \framebox{\(\cos(x + \cos (x + x))\)}\)
\item \(x + \cos (x + x) \rightarrow \framebox{\(x + \cos (x + x)\)}\)
\item \(x + x \rightarrow \framebox{\( 2 x\)}\)
\item \(\cos (\cos (2 x) + (1 - (1 - x))) \rightarrow 
\framebox{\(\cos (\cos (2 x) + (1 - (1 - x)))\)}\)
\item \(\cos (2 x) + (1 - (1 - x)) \rightarrow \framebox{\(x + \cos (2 x)\)}\)
\item \(2 x \rightarrow \framebox{\(2 x\)}\)
\end{itemize}
\item Old equality : \(\cos (x + \cos (x + x)) = \cos (\cos (2 x) + (1 - (1 - x)))\)
\item New equality : \(\cos (x + \cos (2 x)) = \cos (x + \cos (2 x)))\)
\end{itemize}
\end{frame}
\begin{frame} % tp 10
\frametitle{Common structure for {\tt field} and {\tt ring}}
\begin{itemize}
\item Produce ``normal forms'' and then test their equality
\item For {\tt ring}
\begin{itemize}
\item Normal forms are polynomials
\item Equality is syntactic
\end{itemize}
\item For {\tt field}
\begin{itemize}
\item Normal forms are fractions
\item Equality is cross product: \(\frac{p}{q}=\frac{r}{s} \Leftrightarrow p \times s = q \times r \wedge q \neq 0 \wedge s \neq 0\)
\item The equality test of {\tt field} relies on {\tt ring}
\item {\tt field} produces side conditions (non-zero conditions)
\end{itemize}
\end{itemize}
\end{frame}
\begin{frame} % tp 12
\frametitle{Problem with {\tt field\_simplify} for our deep behavior}
\begin{itemize}
\item {\tt field\_simplify} uses normal forms modulo fraction equality
\begin{itemize}
\item cross-product equality
\end{itemize}
\item Reification relies on syntactic equality
\item \(\frac{2 \pi}{4}\) and \(\frac{\pi}{2}\) are equal as fractions
\begin{itemize}
\item \(2 \times 2 \pi = 4 \times \pi\) as verified by {\tt ring}
\end{itemize}
\item {\tt field\_simplify} does not produce the same syntactic expression
\begin{itemize}
\item \(\cos \frac{2 \pi}{4}\) is never transformed into 
\(\cos \frac{\pi}{2}\)
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Justification theorems (stripped down)}
\begin{itemize}
\item Existing theorem for {\tt field\_simplify}\\
\(\forall l~e~r, {\tt Fnorm}~l~e = r \rightarrow
{\tt PCond (condition}~r{\tt )} \rightarrow\)
\strut\hskip1cm \({\tt FEeval}~l~e = {\tt P}_\phi~l~({\tt norm}~({\tt num}~r )) /
   {\tt P}_\phi~l~({\tt norm}~ ({\tt denum}~r))\)
\item For our extension\\
\(\forall l~e~r, {\tt Fnorm}~l~e = r \rightarrow\)\\
\(\forall m~\nu~\delta~\gamma,\)\\
  \( m \neq 0 \rightarrow m\times {\tt num}~r = \nu \times \gamma \rightarrow
  m\times {\tt denum}~r = \delta \times \gamma \rightarrow\) \\
\({\tt PCond (condition}~r{\tt )} \rightarrow\)\\
\({\tt FEeval}~l~e = {\tt P}_\phi~l~ ({\tt norm}~\nu) /
   {\tt P}_\phi~l~({\tt norm}~ \delta)\)

\item We use an oracle to provide the values \(m\), \(\nu\), \(\delta\), and \(\gamma\)
\item The  \(m\) non-zero and common divisor conditions are easy
\end{itemize}
\end{frame}
\begin{frame}
\frametitle{Examples of {\tt Fnorm}, {\tt norm}, {\tt FEeval}}
\begin{itemize}
\item \(1 / (\sin 1 / (\cos 1 + \cos 1))\)
\item Unknown expressions are \(\sin 1\), \(\cos 1\)
\item the ``field'' formula \(e\) is \(1 / (X_1 / (X_2 + X_2)\)
\item {\tt FEeval}~\([\sin 1; \cos 1]\)~(\(1 / (X_1 / (X_2 + X_2))\)) is
 \(1 / (\sin 1 / (\cos 1 + \cos 1))\)
\item \(r =\) {\tt Fnorm}~\(e\) is \((X_2 + X_2) / X_1\) (with conditions)
\item \({\tt norm}~({\tt num}~r) =\) {\tt norm} \((X_2 + X_2)\) is \(2 X_2\)
\item \({\tt P}_\phi~[\sin 1;\cos 1]~(2 X_2)\) is \(2 \cos 1\)
\item Two different types of polynomials need {\tt FEeval} and {\tt P\(_\phi\)}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Encoding and decoding}
\begin{itemize}
\item Rocq has two datatypes to represent polynomials
\begin{itemize}
\item {\tt PExpr} and {\tt Pol}
\item With a function to convert {\tt PExpr} to {\tt Pol} ({\tt norm})
\end{itemize}
\item Elpi has the datatype {\tt polyT}
\item We add a conversion function from {\tt Pol} to {\tt polyT}
\begin{itemize}
\item Complex: {\tt Pol} is a sparse Horner form
\end{itemize}
\item And a conversion function from {\tt PolyT} to {\tt PExpr}
\begin{itemize}
\item Easy: {\tt PExpr} contains a direct mirror of {\tt polyT}
\end{itemize}
\end{itemize}
\end{frame}

\begin{itemize}
