\begin{itemize}

\item Type theory proposes a variety of type numbers

\begin{alltt}
Require Import Reals.
Open Scope R_scope.

Inductive Rnat : R -> Prop :=
  Rnat0 : Rnat 0
| Rnat_succ : forall n, Rnat n -> Rnat (n + 1).
\end{alltt}
Generated induction principle:
\begin{alltt}
nat_ind
     : forall P : R -> Prop,
       P 0 ->
       (forall n : R, Rnat n -> P n -> P (n + 1)) ->
       forall r : R, Rnat r -> P r
\end{alltt}

\begin{frame}[fragile]
\frametitle{Computation}
\begin{itemize}
\item Notations are so that 12 (as a real number) is actually {\tt IZR 13}
\item {\tt IZR~:~Z -> R} is a recursive function
\item The interaction with {\tt Compute} is not pleasant
\end{itemize}
\begin{alltt}
Compute 12.
\textcolor{blue}{ = (R1 + R1) * ((R1 + R1) * (R1 + (R1 + R1)))
     : R}
\end{alltt}
\begin{itemize}
\item We designed a new command called {\tt R\_compute}.
\end{itemize}
\begin{alltt}
R_compute (12 + 13).
\textcolor{blue}{ = 25}
\end{alltt}
\begin{itemize}
\item Does not cover \(\lambda\)-calculus or let-binders
\item Works by using a database of registered functions
\item Can produce a lemma stating the result as an equality
\end{itemize}

\begin{frame}[fragile]
\frametitle{Recursive definitions, following the {\tt nat} recursive scheme}
\begin{itemize}
\item We provide a command that mimicks recursive definitions on {\tt Rnat}
\item Only ``well-defined'' for elements of {\tt Rnat}
\item The command produces two objects
\begin{itemize}
\item The function of type {\tt R -> R}
\item The proof of the logical statement for that function
\end{itemize}
\end{itemize}
\begin{small}
\begin{verbatim}
Recursive (def fib such that
             fib 0 = 0 /\
             fib 1 = 1 /\
             forall n : R, Rnat (n - 2) ->
                fib n = fib (n - 2) + fib (n - 1)).
\end{verbatim}
\end{small}
Note: no pattern matching, but condition {\tt Rnat (n - 2)}
\end{frame}

\begin{frame}
\frametitle{A poor man's {\tt calc} tactic}
\begin{itemize}
\item Lean has popularized a style of proofs called calculational
\item Clever use of transitivity of equality and orders
\item For our experiments, we only cover equality
\item {\tt start\_with} \(formula\)
\begin{itemize}
\item Checks that the goal has the form \(F = G\)
\item Checks that the \(formula\) and \(F\) are convertible
\item Changes the goal into \(formula = G\)
\end{itemize}
\item {\tt calc\_LHS} \(formula\)
\begin{itemize}
\item Checks the goal has the form \(F = G\)
\item Creates two new goals \(F = formula\) and \(formula = F\)
\end{itemize}
\item Should be read as \(... = formula\)
\begin{itemize}
\item The intent is to make proofs declarative (as in {\em waterproof})
\item Each step after one instance of {\tt calc\_LHS} should be solved in an
  easy step
\end{itemize}
\item {\tt end\_calculate} : checks that the intended equality is ``obvious''
\end{itemize}
\end{frame}
\begin{frame}
\frametitle{Supercharging {\tt ring} and {\tt field}}
\begin{itemize}
\item Tactics {\tt ring} and {\tt field} only work from the root
\item Any application of an {\em unknown} function is treated as an atom
\item Arguments are not modified by {\tt ring} or {\tt field}
\item Example {\tt x + sqrt(x + 1) = sqrt (1 + x) + x} is not covered
\item ring treats {\tt sqrt (x + 1)} and {\tt sqrt (1 + x)} as two different atoms
\end{itemize}
\end{frame}
\begin{frame}
\frametitle{A strategy that we propose}
\begin{itemize}
\item Use {\tt ring\_simplify} or {\tt field\_simplify}
\begin{itemize}
\item Normalize formulas given as argument, replace them
\item Can handle several formulas at a time (with the same order)
\end{itemize}
\item normalize expressions under unknown functions
\item repeat until stability
\item On example: normalize {\tt (x + 1)}, {\tt 1 + x}, {\tt x + sqrt (1 + x)}
and {\tt sqrt (1 + x) + x}
\item This may yield {\tt x + sqrt (x + 1) = sqrt (x + 1) + x}
\item Next normalization : {\tt x + sqrt (x + 1) = x + sqrt (x + 1)}
\item We call the new tactics {\tt deep\_ring} and {\tt deep\_field}
\item We also extend {\tt calc\_LHS} to include an attempt to solve with the
  deep tactics
\end{itemize}
\end{frame}
\begin{frame}[fragile]
\frametitle{Example of proof made possible}
\begin{alltt}
Lemma cos_sub_Pi_half x : cos (x - Pi / 2) = sin x.
Proof.
start_with (cos (x - Pi / 2)).
calc_LHS (cos x * cos (Pi / 2) + sin x * sin (Pi / 2)).
  now rewrite cos_sub.
calc_LHS (cos x * 0 + sin x * sin (Pi / 2)).
  now rewrite cos_Pi_half.
calc_LHS (cos x * 0 + sin x * 1).
  now rewrite sin_Pi_half.
end_calculate.
Qed.
\end{alltt}
\end{frame}
\begin{frame}
\frametitle{Formule de Vičte pour le calcul de \(\pi\)}
\[\pi = 2 \cdot \frac 2 {\sqrt{2}} \cdots \frac 2 {\sqrt{2 + \cdots \sqrt{2}}} \cdots\]
\begin{itemize}
\item Use of recursive definition to define \(\sqrt {2 + \sqrt {\cdots + \sqrt 2}}\)
\item Use of iterated products to define the main formula
\item {\bf Key point 1}: \(\sin \frac{\pi}{2 ^ {n + 1}} =  \frac{\sin \frac{\pi}{2 ^ n}}{2 \cos \frac{\pi}{2 ^ {n + 1}}}\)  et \(\sin \frac{\pi}{2} = 1\)
\item {\bf Key point 2}: \(\cos {\frac{\pi}{2 ^ n}} = \frac{\sqrt{2 + \sqrt{... + \sqrt 2}}}{2}\)
\item {\bf Key point 3}: 
  \(\lim_{n -> \infty} \frac{\frac{\pi}{2 ^ n}}{\sin \frac{\pi}{2 ^ n}} = 1\)
\end{itemize}
\end{frame}
